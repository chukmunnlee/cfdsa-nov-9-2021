# create a network
docker network create mynet

# create a volume
docker volume create myvol

# deploy database into mynet
docker run -d --name nwdb --network mynet stackupiss/northwind-db:v1 

# deploy application into mynet
docker run -d -p 8080:3000 \
    --name nwapp --network mynet \
    -e DB_HOST=nwdb \
    -e DB_USER=root \
    -e DB_PASSWORD=changeit \
    stackupiss/northwind-db:v1
